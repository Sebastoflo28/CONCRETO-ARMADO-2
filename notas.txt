    <script>
        function calcularAcero() {
            let fy = parseFloat(document.getElementById('fy').value);
            let fc = parseFloat(document.getElementById('fc').value);
            let b = parseFloat(document.getElementById('b').value);
            let d = parseFloat(document.getElementById('d').value);
            let Mu = parseFloat(document.getElementById('mu').value);
            
            // Fórmula aproximada para el cálculo de As (Área de Acero)
            let As = (Mu) / (0.9 * fy * d);
            
            document.getElementById('resultado').innerText = As.toFixed(2) + ' cm²';
        }
    </script>
